//Inicializar e informar ao git repositorio

->>> git init

//Verificar alterações nos arquivos

->>> git status

//Monitor arquivos no repositorio
//(arquivo unico)

->>> git add nomeArquivo

//(varios arquivos)

->>> git add .

//Comentar codigo

->>> git commit  -m "Criando arquivo index.html com lista de curso"

//enviar para o repositorio na nuvem
--> git push

//modificar cofigurações do git
->>> git config -- local user.name / user.email
->>> git config -- global

//verificar o historicos de modificações no repositorio

->>> git reflog

//Como navegar entre versões do codigo
->> git reset --hard "ID do reflog"
